 45        shape = tableau.shape()

 47        n1 = (-1,-1)
 48        n2 = (-1,-1)
 49

 53                    n1 = (i,j)
 54                if tableau[i, j] == n - 1:
 55                    n2 = (i,j)

130    def _hook_length_prod(self, partition):
131        resp = 1
132        for i in range(partition.height()):
133            for j in range(partition[i]):
134                cells_in_row = (partition[i] - j)
135                cells_below = self._cells_below_cell(partition, i, j)
136                hook_length = (cells_in_row + cells_below)
137                resp = resp * hook_length

146    def _generate_representation_matrix(self, partition, ordered_2ndlev, mode="YKR"):
147        
148        n = len(ordered_2ndlev)
149        matrix = [[[-1,-1,None] for _ in range(n)] for _ in range(n)]
150        
151        # Primero hacemos la diagonal
152        
153        for i in range(n):
154
155            ((a,b),(c,d)) = ordered_2ndlev[i]
156            gamma = self._get_partition_from_gamma((a,c), partition)
157            d_gamma = self._frame_robinson_thrall(gamma)
158            coef = None
159            # Cláusula 3 del teorema de young
160            if a == c: # R-chain
161                coef = 1
162            if b == d: # C-chain
163                coef = -1
164            matrix[i][i] = [d_gamma, d_gamma, coef]
165
166        for i in range(n):
167            for j in range(n):
168                
169                if i < j:
170
171                    ((a,b),(c,d)) = ordered_2ndlev[i]
172                    ((a2,_),(c2,_)) = ordered_2ndlev[j]
173                    gamma = self._get_partition_from_gamma((a,c), partition)
174                    gamma2 = self._get_partition_from_gamma((a2,c2), partition)
175                    d_gamma = matrix[i][i][0]
176                    d_gamma2 = matrix[j][j][0]
177                    
178                    matrix[i][j][0] = d_gamma
179                    matrix[i][j][1] = d_gamma2
180
181                    matrix[j][i][0] = d_gamma2
182                    matrix[j][i][1] = d_gamma
183
184                    if str(gamma) == str(gamma2):
185                        
186                        chi = abs(a-c) + abs(b-d)
187                        chi_sqr = chi**2
188                        q_sqr_chi = Fraction((chi_sqr - 1), chi_sqr)
189                        
190                        if mode == "YKR":
191                            if matrix[i][i][2] == None:
192                                matrix[i][i][2] = Fraction(1,chi)
193                            if matrix[j][j][2] == None:
194                                matrix[j][j][2] = Fraction(-1,chi)
195                            if matrix[i][j][2] == None:
196                                matrix[i][j][2] = Fraction(1)
197                            if matrix[j][i][2] == None:
198                                matrix[j][i][2] = q_sqr_chi
199                            
200                        if mode == "YSR":
201                                if matrix[i][i][2] == None:
202                                    matrix[i][i][2] = Fraction(1,chi)
203                                if matrix[j][j][2] == None:
204                                    matrix[j][j][2] = Fraction(-1,chi)
205                                if matrix[i][j][2] == None:
206                                    matrix[i][j][2] = q_sqr_chi
207                                if matrix[j][i][2] == None:
208                                    matrix[j][i][2] = Fraction(1)
209                        if mode == "YOR":
210                                
211                                sqrt = math.sqrt(q_sqr_chi.numerator/q_sqr_chi.denominator)
212
213                                if matrix[i][i][2] == None:
214                                    matrix[i][i][2] = Fraction(1,chi)
215                                if matrix[j][j][2] == None:
216                                    matrix[j][j][2] = Fraction(-1,chi)
217                                if matrix[i][j][2] == None:
218                                    matrix[i][j][2] = sqrt
219                                if matrix[j][i][2] == None:
220                                    matrix[j][i][2] = sqrt
221                    else:
222                        matrix[i][j][2] = Fraction(0)
223                        matrix[j][i][2] = Fraction(0)
224

243    def _direct_sum(self, matrices):
244        n = len(matrices)
245        
246        total_rows = sum(len(mat) for mat in matrices)
247        total_cols = sum(len(mat[0]) for mat in matrices)
248        
249        result = [[[0, 0, 0] for _ in range(total_cols)] for _ in range(total_rows)]
250        
251        row_start = 0
252        col_start = 0
253        
254        for mat in matrices:
255            mat_rows = len(mat)
256            mat_cols = len(mat[0])
257            
258            for i in range(mat_rows):
259                for j in range(mat_cols):
260                    result[row_start + i][col_start + j] = mat[i][j]
261            
262            row_start += mat_rows
263            col_start += mat_cols
264        
265        for i in range(total_rows):
266            for j in range(total_cols):
267                if result[i][j] == [0, 0, 0]:
268                    result[i][j] = [result[i][i][0], result[j][j][0], 0]
269        
270        return result

272    def _build_irrep_of_transposition(self, partition, t_n: int, mode="YKR"):
273        """
274        Construye la representación irreducible de una transposición (como matriz por bloques).
275
276        Args:
277            partition (Snob2.IntegerPartition): La partición entera asociada.
278            t_n (int): El índice de la transposición.
279            mode (str, optional): El modo de la representación. Debe ser uno de 'YKR', 'YSR' o 'YOR'. Por defecto es 'YKR'.
280        """
281        # Caso base, la partición es de n elementos y queremos evaluar la representación en t_n
282        n = partition.getn()
283        if t_n == n:
284            return self._build_irrep(partition, mode)
285        else:
286            # En otro caso, tenemos que expandir el árbol, es decir
287            # Tenemos que calcular todas las transposiciones cuyos tableros de young salen de la actual
288            partitions = self._branch(partition)
289            submatrix_list = []
290            for part in partitions:
291                submatrix = self._build_irrep_of_transposition(part, t_n, mode)
292                submatrix_list.append(submatrix)
293            resp = self._direct_sum(submatrix_list)
294            return resp

297        """
298        Construye la representación irreducible de una permutación.
299
300        Args:.
301            pi (list): La permutación expresada como una lista de transposiciones adyacentes.
302
303        Returns:
304            np.ndarray: La matriz de la representación irreducible de la permutación.
305        """
306        
307        transpositions = express_into_adyacent_transpositions(pi)
308
309        if self.matrices[transpositions[0]-2] is None:
310            self.matrices[transpositions[0]-2] = self._build_irrep_of_transposition(self.partition, transpositions[0], self.mode)
311        currMatrix = np.array(decompress(self.matrices[transpositions[0]-2]).tolist())
312
313        for transposition in transpositions[1:]:
314            if self.matrices[transposition-2] is None:
315                self.matrices[transposition-2] = self._build_irrep_of_transposition(self.partition, transposition, self.mode)
316            currMatrix = np.matmul(currMatrix, np.array(decompress(self.matrices[transposition-2])).tolist())
317
318        return currMatrix
